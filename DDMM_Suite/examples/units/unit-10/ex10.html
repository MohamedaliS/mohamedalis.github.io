Worked Example 10: The Normal Curve and Z-Scores

Context

You are reviewing examination results for 250 students across three management programmes. Management wants to know whether scores are approximately normally distributed, apply the empirical rule, and compare individual student performance across subjects.

Step 1: Summary statistics

library(dplyr)
score_summary <- df |>
  summarise(
    n    = n(),
    mean = round(mean(statistics_score), 2),
    sd   = round(sd(statistics_score), 2),
    min  = min(statistics_score),
    max  = max(statistics_score)
  )
knitr::kable(score_summary,
  col.names = c("n", "Mean", "SD", "Min", "Max"),
  caption = "Table E10.1: Statistics score summary") |>
  kableExtra::kable_styling(bootstrap_options = c("striped"), full_width = FALSE)

Step 2: Apply the empirical rule

mu <- mean(df$statistics_score)
s  <- sd(df$statistics_score)

emp_check <- data.frame(
  Rule     = c("1 SD", "2 SD", "3 SD"),
  Expected = c("~68%", "~95%", "~99.7%"),
  Actual   = round(100 * c(
    mean(df$statistics_score >= mu - s   & df$statistics_score <= mu + s),
    mean(df$statistics_score >= mu - 2*s & df$statistics_score <= mu + 2*s),
    mean(df$statistics_score >= mu - 3*s & df$statistics_score <= mu + 3*s)
  ), 1)
)
emp_check$Actual <- paste0(emp_check$Actual, "%")
knitr::kable(emp_check,
  caption = "Table E10.2: Empirical rule check for statistics scores") |>
  kableExtra::kable_styling(bootstrap_options = c("striped"), full_width = FALSE)

Step 3: Z-scores for individual students

df_z <- df |>
  mutate(
    z_stats = round((statistics_score - mean(statistics_score)) /
                      sd(statistics_score), 2),
    z_maths = round((maths_score - mean(maths_score)) /
                      sd(maths_score), 2)
  ) |>
  select(student_id, programme, statistics_score, z_stats,
         maths_score, z_maths) |>
  arrange(desc(z_stats)) |>
  head(8)

knitr::kable(df_z,
