Step 2: Categorical frequency distribution

library(dplyr)
cat_dist <- df |>
  count(policy_type, name = "Frequency") |>
  arrange(desc(Frequency)) |>
  mutate(
    Relative  = round(Frequency / sum(Frequency), 3),
    Pct       = paste0(round(100 * Frequency / sum(Frequency), 1), "%")
  )

knitr::kable(cat_dist,
  col.names = c("Policy type", "Frequency", "Relative frequency", "Percentage"),

  kableExtra::kable_styling(bootstrap_options = c("striped", "hover"),

Step 3: Numerical frequency distribution with cumulative frequencies

breaks <- c(0, 2000, 4000, 6000, 8000, 10000, 12000, 14000, 16000)
labels <- c("0-2k", "2-4k", "4-6k", "6-8k", "8-10k", "10-12k", "12-14k", "14-16k")

num_dist <- df |>
  mutate(class = cut(claim_amount_gbp, breaks = breaks,
                     labels = labels, include.lowest = TRUE)) |>
  count(class, name = "Freq", .drop = FALSE) |>
  mutate(
    Rel_freq = round(Freq / sum(Freq), 3),
    Cum_freq = cumsum(Freq),
    Cum_rel  = round(cumsum(Rel_freq), 3)
  )

knitr::kable(num_dist,
  col.names = c("Claim amount (GBP 000s)", "Frequency", "Relative frequency",
                "Cumulative frequency", "Cumulative relative frequency"),
  caption   = "Table E6.3: Claim amount distribution") |>
  kableExtra::kable_styling(bootstrap_options = c("striped", "hover"),
                             full_width = FALSE)
