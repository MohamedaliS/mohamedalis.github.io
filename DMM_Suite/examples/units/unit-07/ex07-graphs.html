Step 2: Bar chart for a categorical variable

library(dplyr)
df |>
  count(product_category) |>
  mutate(product_category = reorder(product_category, n)) |>
  ggplot(aes(x = product_category, y = n)) +
  geom_bar(stat = "identity", fill = "#1B3A5C", alpha = 0.85) +
  coord_flip() +
  labs(x = NULL, y = "Orders") +
  theme_dmm()

Step 3: Histogram for a numerical variable

ggplot(df, aes(x = order_value_gbp)) +
  geom_histogram(bins = 20, fill = "#1B3A5C", colour = "white", alpha = 0.85) +
  labs(x = "Order value (GBP)", y = "Count") +
  theme_dmm()

Step 4: Return rates by category and device

df |>
  group_by(product_category, device_type) |>
  summarise(pct_returned = 100 * mean(returned == "Yes"), .groups = "drop") |>
  ggplot(aes(x = product_category, y = pct_returned, fill = device_type)) +
  geom_bar(stat = "identity", position = "dodge", alpha = 0.85) +
  scale_fill_brewer(palette = "Set2") +
  labs(x = NULL, y = "Return rate (%)", fill = "Device") +
  theme_dmm() +
  theme(axis.text.x = element_text(angle = 30, hjust = 1))

Step 5: Scatter plot

ggplot(df, aes(x = order_value_gbp, y = delivery_days)) +
  geom_point(colour = "#1B3A5C", alpha = 0.55, size = 1.8) +
  geom_smooth(method = "lm", se = FALSE, colour = "#C0392B", linewidth = 0.8) +
  labs(x = "Order value (GBP)", y = "Delivery days") +
  theme_dmm()
