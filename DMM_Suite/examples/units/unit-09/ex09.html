Worked Example 09: Quartiles, Boxplots, and Outlier Detection

Context

You are reviewing asking price data for 180 residential properties. Management wants to understand the price distribution and identify any unusually priced listings.

Step 1: Five-number summary

library(dplyr)
fns <- data.frame(
  Statistic = c("Minimum", "Q1", "Median", "Q3", "Maximum", "IQR"),
  Value = c(
    min(df$asking_price_gbp),
    quantile(df$asking_price_gbp, 0.25),
    median(df$asking_price_gbp),
    quantile(df$asking_price_gbp, 0.75),
    max(df$asking_price_gbp),
    IQR(df$asking_price_gbp)
  )
)
fns$Value <- scales::comma(round(fns$Value, 0))
knitr::kable(fns, col.names = c("Statistic", "Value (GBP)"),
  caption = "Table E9.1: Five-number summary of asking prices") |>
  kableExtra::kable_styling(bootstrap_options = c("striped"), full_width = FALSE)

Step 2: IQR fence rule

q1 <- quantile(df$asking_price_gbp, 0.25)
q3 <- quantile(df$asking_price_gbp, 0.75)
iqr <- IQR(df$asking_price_gbp)
lower_fence <- q1 - 1.5 * iqr
upper_fence <- q3 + 1.5 * iqr

cat("Lower fence: GBP", format(round(lower_fence, 0), big.mark = ","), "\n")
Lower fence: GBP 90,750 
cat("Upper fence: GBP", format(round(upper_fence, 0), big.mark = ","), "\n")
Upper fence: GBP 644,750 
outliers <- df |>
  filter(asking_price_gbp > upper_fence | asking_price_gbp < lower_fence) |>
  arrange(desc(asking_price_gbp)) |>
  select(property_id, property_type, location_tier, bedrooms, asking_price_gbp)

cat("\nOutlier count:", nrow(outliers), "\n\n")

Outlier count: 9

Step 4: Skewness check

skew_check <- df |>
  group_by(property_type) |>
  summarise(
    mean   = round(mean(asking_price_gbp), 0),
    median = round(median(asking_price_gbp), 0),
    shape  = ifelse(mean(asking_price_gbp) > median(asking_price_gbp),
                    "Right-skewed", "Symmetric/left")
  )
knitr::kable(skew_check,
  col.names = c("Type", "Mean (GBP)", "Median (GBP)", "Shape"),
  caption = "Table E9.3: Skewness by property type",
